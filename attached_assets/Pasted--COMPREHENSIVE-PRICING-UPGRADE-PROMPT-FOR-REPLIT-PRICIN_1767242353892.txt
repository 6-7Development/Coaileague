### **C) Update Stripe Webhook Handlers**

```typescript
// When employee count changes, update metered billing
async function handleEmployeeCountChange(
  workspaceId: string, 
  newCount: number
) {
  const subscription = await getSubscription(workspaceId);
  const tier = PRICING_TIERS[subscription.tier];
  
  const overageEmployees = Math.max(0, newCount - tier.includedEmployees);
  
  // Report usage to Stripe for metered billing
  await stripe.subscriptionItems.createUsageRecord(
    subscription.stripeOverageItemId,
    {
      quantity: overageEmployees,
      timestamp: Math.floor(Date.now() / 1000),
      action: 'set' // Replace previous value
    }
  );
  
  // Update local database
  await updateSubscription(workspaceId, {
    employeeCount: newCount,
    overageEmployees,
    overageCharge: overageEmployees * tier.overagePrice,
    totalMonthlyCharge: tier.basePrice + (overageEmployees * tier.overagePrice)
  });
}
```

```typescript
async function convertTrialToP aid(workspaceId: string, selectedTier: string) {
  const tier = PRICING_TIERS.find(t => t.name === selectedTier);
  const workspace = await getWorkspace(workspaceId);
  
  // Calculate initial charge
  const employeeCount = await getEmployeeCount(workspaceId);
  const overageEmployees = Math.max(0, employeeCount - tier.includedEmployees);
  const baseCharge = tier.basePrice;
  const overageCharge = overageEmployees * tier.overagePrice;
  const totalCharge = baseCharge + overageCharge;
  
  // Create Stripe subscription
  const subscription = await stripe.subscriptions.create({
    customer: workspace.stripeCustomerId,
    items: [
      {
        price: tier.stripeBasePriceId, // Base price
      },
      {
        price: tier.stripeOveragePriceId, // Metered employee overage
      }
    ],
    metadata: {
      workspaceId,
      tier: selectedTier,
      employeeCount,
      includedEmployees: tier.includedEmployees
    }
  });
  
  // Report initial overage usage
  if (overageEmployees > 0) {
    await stripe.subscriptionItems.createUsageRecord(
      subscription.items.data[1].id, // Overage item
      { quantity: overageEmployees }
    );
  }
  
  // Update workspace
  await updateWorkspace(workspaceId, {
    subscriptionStatus: 'active',
    tier: selectedTier,
    stripeSubscriptionId: subscription.id,
    trialEnded: new Date()
  });
  
  // Send confirmation email
  await sendEmail({
    to: workspace.ownerEmail,
    subject: `Welcome to CoAIleague ${tier.displayName}!`,
    body: `You're now on the ${tier.displayName} plan.
    
    Your first charge: $${totalCharge}
    Base: $${baseCharge}
    ${overageEmployees > 0 ? `+ ${overageEmployees} employees × $${tier.overagePrice} = $${overageCharge}` : ''}
    
    Trinity AI is now fully activated and managing your workforce.
    
    Next billing date: ${getNextBillingDate()}
    `
  });
}
```
