            function setProgressRadial(progressId, percent) {
                const progressCircle = document.getElementById(progressId);
                const offset = CIRCUMFERENCE - (percent / 100) * CIRCUMFERENCE;
                progressCircle.style.strokeDashoffset = offset;
            }

            // --- 1. Force Flow (Linear Progress) ---
            function startForceFlow() {
                const fill = document.getElementById('force-flow-fill');
                const percentDisplay = document.getElementById('force-flow-percent');
                const messageDisplay = document.getElementById('force-flow-message');
                let interval;

                // Reset
                clearInterval(interval);
                fill.style.width = '0%';
                percentDisplay.textContent = '0%';
                messageDisplay.textContent = 'Initializing Force Conduit...';

                let progress = 0;
                interval = setInterval(() => {
                    progress += Math.floor(Math.random() * 5) + 1; // Random increment
                    
                    if (progress >= 100) {
                        progress = 100;
                        clearInterval(interval);
                        messageDisplay.textContent = 'FORCE CONDUIT FLOW COMPLETE. STATUS: OPTIMAL.';
                    }

                    fill.style.width = `${progress}%`;
                    percentDisplay.textContent = `${progress}%`;

                    if (progress > 85 && progress < 100) {
                        messageDisplay.textContent = 'Optimizing final throughput...';
                    }
                }, 100);
            }

                const animatedA = document.getElementById('animated-A');
                const completionF = document.getElementById('completion-F');
                let interval;

                // Reset
                clearInterval(interval);
                setProgressRadial('af-core-progress', 0);
                percentDisplay.textContent = '0%';
                messageDisplay.textContent = 'SCANNING AUTO-FORCE CORE...';
                
                // Reset F and start A animation
                animatedA.classList.add('af-core-a-spinning');
                animatedA.style.transform = 'translateX(-1.2rem)'; // Ensure A is positioned correctly
                completionF.classList.add('opacity-0', 'scale-50');
                completionF.classList.remove('opacity-100', 'scale-100');
                
                let progress = 0;

                interval = setInterval(() => {
                    progress += Math.floor(Math.random() * 3) + 1;
                    
                    if (progress >= 100) {
                        progress = 100;
                        clearInterval(interval);
                        
                        // Completion Actions
                        messageDisplay.textContent = 'AF CORE: SECURE FORCE LOCK ESTABLISHED.';
                        
                        // Stop A spin and center it slightly to the left
                        animatedA.classList.remove('af-core-a-spinning'); 
                        animatedA.style.transform = 'translateX(-1.2rem) scale(1.1)'; // Slight scale for effect
                        
                        // Reveal the F with animation
                        completionF.classList.remove('opacity-0', 'scale-50');
                        completionF.classList.add('opacity-100', 'scale-100');

                        // Set final visual state
                        setProgressRadial('af-core-progress', 100);
                        percentDisplay.textContent = '100%';

                    } else {
                        // Maintain the correct position during spin
                        animatedA.style.transform = `translateX(-1.2rem) rotateY(${progress * 5}deg)`; 
                    }

                    setProgressRadial('af-core-progress', progress);
                    percentDisplay.textContent = `${progress}%`;

                    if (progress > 50 && progress < 80) {
                        messageDisplay.textContent = 'ANALYZING CORE INTEGRITY... SECTOR GREEN.';
                    } else if (progress >= 80 && progress < 100) {
                        messageDisplay.textContent = 'SYNTHESIZING FORCE MATRIX... FINALIZING.';
                    }

                }, 50);
            }

                const status = document.getElementById('data-stream-status');
                const message = document.getElementById('data-stream-message');
                const text = document.getElementById('data-stream-text');
                let interval;

                // Reset
                clearInterval(interval);
                fill.style.height = '0%'; // Start at 0
                status.textContent = 'READY';
                status.classList.remove('text-cyan-400');
                status.classList.add('text-purple-400');
                text.textContent = '0% LOADED';
                text.style.color = 'var(--color-text)';

                const totalLoadTime = 6000; // 6 seconds for a smooth fill
                const startTime = Date.now();

                // Start wave animation and update status
                status.classList.add('text-cyan-400');
                status.classList.remove('text-purple-400');

                    progress = Math.min(100, Math.round((elapsed / totalLoadTime) * 100));

                        status.textContent = 'COMPLETE';

                        text.style.color = 'var(--color-bg)';
                    } else if (progress > 50) {

                        text.style.color = 'var(--color-bg)';
                    } else if (progress > 10) {

                    }

            const hexCells = [];
            const hexGridEl = document.getElementById('hex-grid');

            function initializeHexGrid() {
                hexGridEl.innerHTML = '';
                hexCells.length = 0; // Clear array
                for (let i = 0; i < HEX_COUNT; i++) {
                    const cell = document.createElement('div');
                    cell.className = 'hex-cell';
                    cell.dataset.index = i;
                    hexCells.push(cell);
                    hexGridEl.appendChild(cell);
                }
            }
            
            function startHexLoader() {
                const message = document.getElementById('hex-loader-message');
                const button = document.getElementById('start-hex-loader');
                
                // Reset
                clearInterval(hexInterval);
                hexCells.forEach(cell => {
                    cell.classList.remove('hex-active', 'hex-fade');
                });
                message.textContent = 'INITIATING SCAN CYCLE...';
                button.disabled = true;

                let currentIndex = 0;
                let cycleCount = 0;
                const maxCycles = 4; 

                hexInterval = setInterval(() => {
                    
                    let prevIndex = currentIndex - 1;
                    if (prevIndex < 0) {
                        prevIndex = HEX_COUNT - 1; 
                    }

                    // De-activate and start the fade-out process on the previous cell
                    if (cycleCount > 0 || currentIndex > 0) {
                        hexCells[prevIndex].classList.remove('hex-active');
                        hexCells[prevIndex].classList.add('hex-fade'); 
                    }
                    
                    // Boundary check for start of a new cycle
                    if (currentIndex >= HEX_COUNT) {
                        currentIndex = 0;
                        cycleCount++;
                        if (cycleCount >= maxCycles) {
                            // Final clean up and stop
                            clearInterval(hexInterval);
                            message.textContent = 'HEX SCAN CYCLE COMPLETE. DATA MATRIX LOCKED.';
                            button.disabled = false;
                            // Reset state after fade out
                            setTimeout(() => {
                                 hexCells.forEach(cell => cell.classList.remove('hex-fade'));
                            }, 1500); 
                            return;
                        }
                    }
                    
                    // Activate the current cell
                    hexCells[currentIndex].classList.remove('hex-fade'); 
                    hexCells[currentIndex].classList.add('hex-active');
                    message.textContent = `ANALYZING GRID CELL ${currentIndex + 1}/${HEX_COUNT} (CYCLE ${cycleCount + 1}/${maxCycles})`;
                    currentIndex++;
                    
                }, 80); // Adjusted pulse rate for sophistication
            }


            // --- 3. Protocol Pipeline (Stepped Progress) ---
            let currentStep = 1;
            
            function resetPipeline() {
                currentStep = 1;
                const button = document.getElementById('pipeline-button');
                button.disabled = false;
                button.textContent = 'START PROTOCOL PIPELINE';
                
                button.removeEventListener('click', resetPipeline); 
                button.addEventListener('click', startProtocolPipeline);

                for (let i = 1; i <= PIPELINE_MESSAGES.length; i++) {
                    const dot = document.getElementById(`step-dot-${i}`);
                    dot.className = 'step-dot step-pending';
                    dot.textContent = i;
                    
                    const msgEl = document.getElementById(`step-message-${i}`);
                    msgEl.textContent = PIPELINE_MESSAGES[i - 1];
                    msgEl.classList.remove('text-orange-400', 'text-cyan-400');
                    msgEl.classList.add('text-gray-400');
                    
                    if (i < PIPELINE_MESSAGES.length) {
                        const connectorFill = document.getElementById(`connector-${i}-fill`);
                        if (connectorFill) {
                            connectorFill.style.height = '0%';
                        }
                    }
                }
            }

            function advancePipeline() {
                const button = document.getElementById('pipeline-button');
                
                if (currentStep > PIPELINE_MESSAGES.length) {
                    button.textContent = 'PROTOCOL COMPLETE / RESET';
                    button.removeEventListener('click', startProtocolPipeline);
                    button.addEventListener('click', resetPipeline);
                    button.disabled = false;
                    return;
                }

                const isLastStep = (currentStep === PIPELINE_MESSAGES.length);

                // 1. Mark previous step (if any) as complete
                if (currentStep > 1) {
                    const prevStepIndex = currentStep - 1;
                    const prevDot = document.getElementById(`step-dot-${prevStepIndex}`);
                    prevDot.className = 'step-dot step-complete';
                    prevDot.innerHTML = '&#10003;';
                    document.getElementById(`step-message-${prevStepIndex}`).classList.add('text-cyan-400');
                    document.getElementById(`step-message-${prevStepIndex}`).classList.remove('text-orange-400', 'text-gray-400');
                    
                    const connectorIndexToFill = prevStepIndex;
                    const connectorFill = document.getElementById(`connector-${connectorIndexToFill}-fill`);
                    if (connectorFill) {
                        connectorFill.style.height = '100%';
                    }
                }

                // 2. Mark current step as active
                const currentDot = document.getElementById(`step-dot-${currentStep}`);
                currentDot.className = 'step-dot step-active';
                document.getElementById(`step-message-${currentStep}`).textContent = PIPELINE_MESSAGES[currentStep - 1] + ' (ACTIVE)';
                document.getElementById(`step-message-${currentStep}`).classList.add('text-orange-400');
                document.getElementById(`step-message-${currentStep}`).classList.remove('text-gray-400');
                
                
                // 3. Prepare for next step
                currentStep++;

                if (!isLastStep) {
                    setTimeout(advancePipeline, STEP_DELAY);
                } else {
                    setTimeout(() => {
                        currentDot.className = 'step-dot step-complete';
                        currentDot.innerHTML = '&#10003;'; 
                        advancePipeline(); // Triggers the completion/reset state
                    }, STEP_DELAY);
                }
            }

            function startProtocolPipeline() {
                const button = document.getElementById('pipeline-button');
                button.disabled = true;
                button.textContent = 'PROTOCOL RUNNING...';
                button.removeEventListener('click', startProtocolPipeline);
                advancePipeline();
            }

            // --- Initial Event Listener Setup ---
            document.getElementById('start-force-flow').addEventListener('click', startForceFlow);
            document.getElementById('start-af-core-scan').addEventListener('click', startAfCoreScan);
            document.getElementById('start-data-stream').addEventListener('click', startDataStream);
            document.getElementById('start-hex-loader').addEventListener('click', startHexLoader);

            // --- Initial Load ---
            initializeHexGrid();
            resetPipeline();
        });
