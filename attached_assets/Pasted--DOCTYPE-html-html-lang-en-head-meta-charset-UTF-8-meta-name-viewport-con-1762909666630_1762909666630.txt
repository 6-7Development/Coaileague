        function togglePassword() {
            const input = document.getElementById('passwordInput');
            const btn = document.querySelector('.toggle-password');
            
            if (input.type === 'password') {
                input.type = 'text';
                btn.textContent = '🔒';
            } else {
                input.type = 'password';
                btn.textContent = '👁️';
            }
        }

        function handleSubmit(event) {

        }
