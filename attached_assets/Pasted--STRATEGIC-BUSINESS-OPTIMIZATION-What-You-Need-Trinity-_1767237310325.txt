typescriptfunction calculateEmployeeScore(employee: Employee): number {
  // Weighted composite score
  const reliabilityWeight = 0.40; // 40% - most important
  const satisfactionWeight = 0.30; // 30% - client retention
  const experienceWeight = 0.15; // 15% - expertise
  const attendanceWeight = 0.15; // 15% - consistency
  
  const reliabilityScore = calculateReliability(employee);
  const satisfactionScore = calculateSatisfaction(employee);
  const experienceScore = calculateExperience(employee);
  const attendanceScore = employee.attendanceRate;
  
  return (
    reliabilityScore * reliabilityWeight +
    satisfactionScore * satisfactionWeight +
    experienceScore * experienceWeight +
    attendanceScore * attendanceWeight
  );
}

function calculateReliability(employee: Employee): number {
  const totalIssues = employee.noShows + employee.callIns + employee.lateArrivals;
  const issueRate = totalIssues / Math.max(employee.totalShiftsAssigned, 1);
  
  // Penalize heavily for reliability issues
  const baseScore = 100 - (issueRate * 100);
  
  // Extra penalty for no-shows (worse than call-ins)
  const noShowPenalty = employee.noShows * 5;
  
  return Math.max(0, baseScore - noShowPenalty);
}

function calculateSatisfaction(employee: Employee): number {
  const totalFeedback = employee.clientComplaints + employee.clientPraise;
  if (totalFeedback === 0) return 75; // Neutral score for no data
  
  const positiveRatio = employee.clientPraise / totalFeedback;
  const complaintPenalty = employee.clientComplaints * 3;
  
  return Math.max(0, Math.min(100, (positiveRatio * 100) - complaintPenalty));
}

typescriptfunction calculateClientTier(client: Client): ClientTier {
  const score = 
    (client.monthlyRevenue / 1000) * 0.30 + // Revenue weight
    client.yearsAsClient * 10 * 0.25 + // Loyalty weight
    client.satisfactionScore * 0.20 + // Satisfaction weight
    client.averageProfitMargin * 0.15 + // Profitability weight
    (client.renewalProbability) * 0.10; // Retention risk weight
  
  if (score >= 80) return { tier: 'enterprise', tierScore: score };
  if (score >= 60) return { tier: 'premium', tierScore: score };
  if (score >= 40) return { tier: 'standard', tierScore: score };
  return { tier: 'trial', tierScore: score };
}

typescriptclass ProfitOptimizationService {
  calculateShiftProfit(shift: Shift, employee: Employee): ShiftProfitMetrics {
    // Base calculation
    const billableRate = shift.billableRate;
    const employeeCost = employee.effectiveCostPerHour; // includes taxes, overhead
    const profitPerHour = billableRate - employeeCost;
    const totalProfit = profitPerHour * shift.durationHours;
    const profitMargin = (profitPerHour / billableRate) * 100;
    
    // Distance/commute cost
    const distance = calculateDistance(
      employee.homeAddress,
      shift.location
    );
    const estimatedFuelCost = distance * 0.67; // IRS mileage rate
    const netProfit = totalProfit - estimatedFuelCost;
    
    // Risk adjustment
    const riskFactor = this.calculateRiskFactor(employee, shift.client);
    const riskAdjustedProfit = netProfit * (1 - riskFactor);
    
    return {
      billableRate,
      employeeCost,
      profitPerHour,
      totalProfit,
      profitMargin,
      commuteMiles: distance,
      estimatedFuelCost,
      netProfit,
      riskFactor,
      riskAdjustedProfit,
      recommendation: this.getRecommendation(riskAdjustedProfit, profitMargin)
    };
  }
  
  calculateRiskFactor(employee: Employee, client: Client): number {
    let risk = 0;
    
    // Employee reliability risk
    if (employee.noShows > 0) risk += 0.05 * employee.noShows;
    if (employee.callIns > 2) risk += 0.02 * employee.callIns;
    if (employee.clientComplaints > 0) risk += 0.10 * employee.clientComplaints;
    
    // Client value risk (higher risk of losing valuable clients)
    if (client.tier === 'enterprise' && employee.overallScore < 85) risk += 0.15;
    if (client.isAtRisk && employee.clientSatisfactionScore < 90) risk += 0.20;
    
    // Commute risk
    const distance = calculateDistance(employee.homeAddress, client.location);
    if (distance > 20) risk += 0.10;
    if (distance > 30) risk += 0.20;
    
    return Math.min(0.5, risk); // Cap at 50% risk
  }
}
