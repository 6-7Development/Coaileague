        // Show context menu
        function showContextMenu(event, userData) {
            event.preventDefault();
            currentContextUser = userData;

            const menuItems = state.isAgent ? agentMenuItems : userMenuItems;
            
            contextMenu.innerHTML = menuItems.map(section => `
                <div class="menu-section">
                    ${section.items.map(item => `
                        <div class="menu-item ${item.danger ? 'danger' : ''}" data-action="${item.action}">
                            <div class="menu-icon">${item.icon}</div>
                            <div>${item.label}</div>
                        </div>
                    `).join('')}
                </div>
            `).join('');

            // Position the menu
            const x = Math.min(event.clientX, window.innerWidth - 240);
            const y = Math.min(event.clientY, window.innerHeight - 400);
            
            contextMenu.style.left = x + 'px';
            contextMenu.style.top = y + 'px';
            contextMenu.classList.add('show');

            // Add click handlers
            contextMenu.querySelectorAll('.menu-item').forEach(item => {
                item.addEventListener('click', () => {
                    const action = item.dataset.action;
                    handleMenuAction(action, currentContextUser);
                    hideContextMenu();
                });
            });
        }

        // Hide context menu
        function hideContextMenu() {
            contextMenu.classList.remove('show');
            currentContextUser = null;
        }

        // Handle menu actions
        function handleMenuAction(action, userData) {
            switch(action) {
                case 'dm':
                    showModal('Send Direct Message', createDMForm(userData));
                    break;
                case 'viewProfile':
                    showModal('User Profile', createProfileView(userData));
                    break;
                case 'ticketHistory':
                    showModal('Ticket History', createTicketHistory(userData));
                    break;
                case 'assignTicket':
                    showModal('Assign Ticket', createAssignTicketForm(userData));
                    break;
                case 'addNote':
                    showModal('Add Note', createAddNoteForm(userData));
                    break;
                case 'addTag':
                    showModal('Add Tag', createAddTagForm(userData));
                    break;
                case 'setStatus':
                    showModal('Set Status', createSetStatusForm(userData));
                    break;
                case 'setPriority':
                    showModal('Set Priority', createSetPriorityForm(userData));
                    break;
                case 'mute':
                    if (confirm(`Mute ${userData.name}?`)) {
                        showToast('success', `${userData.name} has been muted`);
                    }
                    break;
                case 'flag':
                    showModal('Flag User', createFlagUserForm(userData));
                    break;
                case 'ban':
                    showModal('Ban User', createBanUserForm(userData));
                    break;
                case 'mention':
                    document.getElementById('messageInput').value += `@${userData.name} `;
                    document.getElementById('messageInput').focus();
                    break;
                case 'block':
                    if (confirm(`Block ${userData.name}?`)) {
                        showToast('success', `${userData.name} has been blocked`);
                    }
                    break;
            }
        }

        function showToast(type, message) {
            const toast = document.createElement('div');
            toast.className = `toast toast-${type}`;
            
            const icons = {
                success: '✓',
                error: '✕',
                info: 'ℹ'
            };
            
            toast.innerHTML = `
                <div class="toast-icon">${icons[type]}</div>
                <div class="toast-message">${message}</div>
            `;
            
            document.getElementById('toastContainer').appendChild(toast);
            
            setTimeout(() => {
                toast.style.animation = 'toastSlideIn 0.3s ease reverse';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }

        // Event listeners
        document.addEventListener('DOMContentLoaded', () => {
            // Add right-click to user items
            document.querySelectorAll('.user-item').forEach(item => {
                item.addEventListener('contextmenu', (e) => {
                    const userData = {
                        name: item.dataset.user,
                        role: item.dataset.role,
                        status: item.dataset.status
                    };
                    showContextMenu(e, userData);
                });
            });

            // Add right-click to message authors
            document.querySelectorAll('.message-author').forEach(author => {
                author.addEventListener('contextmenu', (e) => {
                    const userData = {
                        name: author.dataset.user,
                        role: author.dataset.role,
                        status: 'online'
                    };
                    showContextMenu(e, userData);
                });
            });

            // Click outside to close context menu
            document.addEventListener('click', (e) => {
                if (!contextMenu.contains(e.target)) {
                    hideContextMenu();
                }
                
                // Close macro menu when clicking outside
                const macroBtn = document.getElementById('macroBtn');
                const macroMenu = document.getElementById('macroMenu');
                if (macroMenu && !macroBtn.contains(e.target) && !macroMenu.contains(e.target)) {
                    macroMenu.classList.remove('show');
                }
            });

            // Modal overlay click
            document.getElementById('modalOverlay').addEventListener('click', (e) => {
                if (e.target.id === 'modalOverlay') {
                    hideModal();
                }
            });

            // Send message
            const sendButton = document.getElementById('sendButton');
            const messageInput = document.getElementById('messageInput');

            sendButton.addEventListener('click', sendMessage);
            messageInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    sendMessage();
                }
            });
        });

        function sendMessage() {
            const input = document.getElementById('messageInput');
            const message = input.value.trim();
            
            if (message) {
                const messagesContainer = document.getElementById('messagesContainer');
                const time = new Date().toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' });
                
                const messageDiv = document.createElement('div');
                messageDiv.className = 'message';
                messageDiv.innerHTML = `
                    <div class="message-header">
                        <div class="message-author" data-user="${state.currentUser}" data-role="agent">${state.currentUser}</div>
                        <div class="message-time">${time}</div>
                    </div>
                    <div class="message-content">${message}</div>
                `;
                
                messagesContainer.appendChild(messageDiv);
                messagesContainer.scrollTop = messagesContainer.scrollHeight;
                
                // Add context menu to new message author
                const newAuthor = messageDiv.querySelector('.message-author');
                newAuthor.addEventListener('contextmenu', (e) => {
                    const userData = {
                        name: newAuthor.dataset.user,
                        role: newAuthor.dataset.role,
                        status: 'online'
                    };
                    showContextMenu(e, userData);
                });
                
                input.value = '';
            }
        }
