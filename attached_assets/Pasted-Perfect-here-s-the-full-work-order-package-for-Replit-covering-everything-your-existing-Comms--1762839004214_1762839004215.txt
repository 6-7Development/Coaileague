💬 2. Chatroom + Room Management Routes

Create / Join Room

routes/rooms.ts

import { Router } from "express";
import { Pool } from "pg";
import crypto from "crypto";
const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const r = Router();

r.post("/create", async (req:any,res)=>{
  const { title, schedule_id, work_assignment_id } = req.body;
  const token = crypto.randomBytes(6).toString("hex");
  const { rows } = await pool.query(
    "INSERT INTO rooms (org_id, token, title, schedule_id, work_assignment_id, owner_user_id, status, visibility, created_by) VALUES ($1,$2,$3,$4,$5,$6,'open','org',$6) RETURNING *",
    [req.user.org_id, token, title||"Untitled Room", schedule_id, work_assignment_id, req.user.id]
  );
  res.json({ room: rows[0], url: `/r/${rows[0].token}` });
});
export default r;

📎 3. File Upload + Attachment Handling

routes/uploads.ts

import { Router } from "express";
import multer from "multer";
import fs from "fs";
const upload = multer({ dest: "uploads/" });
const r = Router();

r.post("/", upload.single("file"), (req:any,res:any)=>{
  if(!req.file) return res.status(400).json({error:"no_file"});
  const meta = {
    id: crypto.randomUUID(),
    uploader_id: req.user.id,
    org_id: req.user.org_id,
    room_id: req.body.room_id,
    filename: req.file.originalname,
    path: req.file.path,
    mimetype: req.file.mimetype,
    size: req.file.size
  };
  // TODO: insert into uploads table
  res.json({ ok:true, meta });
});
export default r;

🕑 6. Room Lifecycle + Archiving

Auto-close job (daily cron):

node scripts/autoCloseRooms.js

scripts/autoCloseRooms.js

import { Pool } from "pg";
import fs from "fs";
const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const rooms = await pool.query(
  "SELECT * FROM rooms WHERE status='open' AND auto_close_at <= now()"
);
for (const room of rooms.rows) {
  await pool.query("UPDATE rooms SET status='closed' WHERE id=$1", [room.id]);
  const messages = await pool.query("SELECT * FROM messages WHERE room_id=$1", [room.id]);
  fs.writeFileSync(`archives/${room.token}.json`, JSON.stringify(messages.rows));
  await pool.query(
    "INSERT INTO room_events (room_id, actor_id, event, payload) VALUES ($1,$2,'archive',$3)",
    [room.id, room.owner_user_id, { export_path: `/archives/${room.token}.json` }]
  );
}
process.exit();
