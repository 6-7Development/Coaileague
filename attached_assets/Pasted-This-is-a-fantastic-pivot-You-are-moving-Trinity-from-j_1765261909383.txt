The Orchestration Function
async function guruOrchestrator(userRequest, context) {
  // 1. Initial Thought & Triage
  const diagnosis = await gemini.generate({
    prompt: `Analyze this issue: "${userRequest}". 
             User Context: ${JSON.stringify(context)}. 
             Output: JSON { intent, severity, suggested_tools }`
  });

  // 2. Tool Execution Loop
  let currentStatus = diagnosis;
  let toolOutputs = [];

  for (const tool of diagnosis.suggested_tools) {
    console.log(`Trinity: Using tool ${tool.name}...`);
    const output = await executeTool(tool.name, tool.params);
    toolOutputs.push(output);
    
    // 3. Re-evaluate based on tool output
    const reflection = await gemini.generate({
      prompt: `Tool ${tool.name} returned: ${output}. Does this solve the issue?`
    });
    
    if (reflection.solved) break;
  }

  // 4. Sub-Agent Dispatch (if needed)
  if (!reflection.solved) {
     return await dispatchToSubAgent("Mechanic", { logs: toolOutputs });
  }

  // 5. Final Response
  return generateUserResponse(toolOutputs);
}
