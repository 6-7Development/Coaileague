import { useEffect } from 'react';
import { useRouter } from 'next/navigation';

export function useNavigationGuard(hasUnsavedChanges: boolean) {
  const router = useRouter();
  
  useEffect(() => {
    const handleBeforeUnload = (e: BeforeUnloadEvent) => {
      if (hasUnsavedChanges) {
        e.preventDefault();
        e.returnValue = 'You have unsaved changes. Are you sure you want to leave?';
      }
    };
    
    window.addEventListener('beforeunload', handleBeforeUnload);
    
    return () => {
      window.removeEventListener('beforeunload', handleBeforeUnload);
    };
  }, [hasUnsavedChanges]);
  
  const guardedNavigate = (path: string) => {
    if (hasUnsavedChanges) {
      const confirmed = confirm('You have unsaved changes. Discard them?');
      if (!confirmed) return;
    }
    router.push(path);
  };
  
  return { guardedNavigate };
}

typescript// src/middleware/rateLimiter.ts
import { Ratelimit } from '@upstash/ratelimit';
import { Redis } from '@upstash/redis';

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!
});

export class SmartRateLimiter {
  private aiRateLimiter: Ratelimit;
  private apiRateLimiter: Ratelimit;
  
  constructor() {
    this.aiRateLimiter = new Ratelimit({
      redis,
      limiter: Ratelimit.slidingWindow(100, '1 h'), // 100 AI calls per hour
      analytics: true
    });
    
    this.apiRateLimiter = new Ratelimit({
      redis,
      limiter: Ratelimit.slidingWindow(1000, '1 h'), // 1000 API calls per hour
      analytics: true
    });
  }
  
  async checkAILimit(workspaceId: string): Promise<{ allowed: boolean; remaining: number }> {
    const { success, limit, remaining, reset } = await this.aiRateLimiter.limit(
      `ai:${workspaceId}`
    );
    
    if (!success) {
      // Log rate limit hit
      await this.logRateLimitHit(workspaceId, 'ai');
      
      // Notify admin
      await this.notifyRateLimitHit(workspaceId, 'AI operations');
    }
    
    return { allowed: success, remaining };
  }
  
  async checkAPILimit(workspaceId: string, partner: string): Promise<{ allowed: boolean; remaining: number }> {
    const { success, limit, remaining } = await this.apiRateLimiter.limit(
      `api:${workspaceId}:${partner}`
    );
    
    if (!success) {
      await this.logRateLimitHit(workspaceId, partner);
      await this.notifyRateLimitHit(workspaceId, `${partner} API calls`);
    }
    
    return { allowed: success, remaining };
  }
  
  // Tier-based limits
  async getTierLimits(tier: string): Promise<RateLimits> {
    const limits = {
      free: { ai: 50, api: 500 },
      starter: { ai: 200, api: 2000 },
      professional: { ai: 1000, api: 10000 },
      enterprise: { ai: 10000, api: 100000 }
    };
    
    return limits[tier] || limits.free;
  }
}

// Usage in API routes
export async function POST(req: Request) {
  const { workspaceId } = await req.json();
  const rateLimiter = new SmartRateLimiter();
  
  // Check rate limit
  const { allowed, remaining } = await rateLimiter.checkAILimit(workspaceId);
  
  if (!allowed) {
    return NextResponse.json(
      { 
        error: 'Rate limit exceeded',
        message: 'You\'ve reached your hourly AI limit. Please try again later or upgrade your plan.',
        remaining: 0
      },
      { status: 429 }
    );
  }
  
  // Proceed with AI operation
  const result = await geminiClient.generateSchedule(data);
  
  // Return with rate limit headers
  return NextResponse.json(result, {
    headers: {
      'X-RateLimit-Remaining': remaining.toString(),
      'X-RateLimit-Limit': '100'
    }
  });
}
3.2 Cost Caps & Budget Alerts
typescript// src/lib/costProtection.ts
export class CostProtectionSystem {
  async checkBudgetLimit(workspaceId: string): Promise<{ allowed: boolean; reason?: string }> {
    const workspace = await db.workspace.findUnique({
      where: { id: workspaceId },
      include: { subscription: true }
    });
    
    const usage = await this.getCurrentMonthUsage(workspaceId);
    const budget = workspace.monthlyBudgetCap || Infinity;
    
    // Check if already over budget
    if (usage.totalCost >= budget) {
      return {
        allowed: false,
        reason: `Monthly budget cap of $${budget} reached`
      };
    }
    
    // Check if this operation would exceed budget
    const estimatedCost = await this.estimateOperationCost(workspaceId);
    if (usage.totalCost + estimatedCost > budget) {
      return {
        allowed: false,
        reason: `This operation would exceed your monthly budget cap`
      };
    }
    
    // Check tier limits
    const tierLimit = this.getTierMonthlyLimit(workspace.subscription.tier);
    if (usage.totalCost >= tierLimit) {
      return {
        allowed: false,
        reason: `Monthly limit for ${workspace.subscription.tier} tier reached`
      };
    }
    
    return { allowed: true };
  }
  
  async setBudgetCap(workspaceId: string, monthlyCap: number) {
    await db.workspace.update({
      where: { id: workspaceId },
      data: { monthlyBudgetCap: monthlyCap }
    });
    
    // Create budget alert
    await db.budgetAlert.create({
      data: {
        workspaceId,
        type: 'budget_cap_set',
        amount: monthlyCap,
        message: `Monthly budget cap set to $${monthlyCap}`
      }
    });
  }
  
  async enableCostAlerts(workspaceId: string, thresholds: number[]) {
    // thresholds = [50, 75, 90, 100] (percentages)
    for (const threshold of thresholds) {
      await db.costAlert.create({
        data: {
          workspaceId,
          threshold,
          enabled: true
        }
      });
    }
  }
}

export class NotificationSystem {
  async sendUsageAlert(workspaceId: string, alert: UsageAlert) {
    const workspace = await db.workspace.findUnique({
      where: { id: workspaceId },
      include: { owner: true, notificationSettings: true }
    });
    
    const channels = workspace.notificationSettings.enabledChannels;
    
    // Email
    if (channels.includes('email')) {
      await this.sendEmail(workspace.owner.email, {
        subject: `⚠️ ${alert.title}`,
        template: 'usage-alert',
        data: {
          workspaceName: workspace.name,
          message: alert.message,
          currentUsage: alert.currentUsage,
          limit: alert.limit,
          actionUrl: `${process.env.APP_URL}/dashboard/usage`
        }
      });
    }
    
    // In-App
    if (channels.includes('in_app')) {
      await db.notification.create({
        data: {
          workspaceId,
          type: 'usage_alert',
          title: alert.title,
          message: alert.message,
          severity: alert.severity,
          actionUrl: '/dashboard/usage',
          read: false
        }
      });
    }
    
    // Slack
    if (channels.includes('slack') && workspace.slackWebhook) {
      await this.sendSlackMessage(workspace.slackWebhook, {
        text: `⚠️ *${alert.title}*`,
        blocks: [
          {
            type: 'section',
            text: {
              type: 'mrkdwn',
              text: alert.message
            }
          },
          {
            type: 'actions',
            elements: [
              {
                type: 'button',
                text: { type: 'plain_text', text: 'View Usage' },
                url: `${process.env.APP_URL}/dashboard/usage`
              }
            ]
          }
        ]
      });
    }
    
    // SMS (for critical alerts only)
    if (alert.severity === 'critical' && channels.includes('sms')) {
      await this.sendSMS(workspace.owner.phone, 
        `AutoForce Alert: ${alert.title}. ${alert.message}`
      );
    }
  }
}
