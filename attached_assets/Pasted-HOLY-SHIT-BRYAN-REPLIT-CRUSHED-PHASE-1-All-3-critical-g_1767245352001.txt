async function rollbackSchedule(scheduleId, reason, notifyEmployees) {
  // 1. Validate permissions
  if (!user.isAdmin && !user.isManager) {
    throw new Error('Unauthorized');
  }
  
  // 2. Get most recent snapshot
  const snapshot = await db.schedule_snapshots.findOne({
    scheduleId: { $ne: scheduleId }, // Previous schedule, not current
    order: { createdAt: 'DESC' }
  });
  
  if (!snapshot) {
    throw new Error('No previous version to rollback to');
  }
  
  // 3. Get affected employees (compare current vs snapshot)
  const currentSchedule = await getSchedule(scheduleId);
  const affectedEmployees = getAffectedEmployees(
    currentSchedule.shifts,
    snapshot.snapshotData.shifts
  );
  
  // 4. Restore previous schedule
  await db.schedules.update(scheduleId, {
    status: 'rolled_back',
    rolledBackAt: new Date(),
    rolledBackBy: userId,
    rolledBackReason: reason
  });
  
  // Restore snapshot as new active schedule
  await db.schedules.create({
    workspaceId: currentSchedule.workspaceId,
    status: 'published',
    shifts: snapshot.snapshotData.shifts,
    restoredFrom: snapshot.id,
    publishedAt: new Date()
  });
  
  // 5. Audit trail
  await db.audit_trail.create({
    action: 'schedule.rollback',
    userId,
    details: {
      scheduleId,
      snapshotId: snapshot.id,
      reason,
      affectedEmployees: affectedEmployees.length
    }
  });
  
  // 6. Notify affected employees
  if (notifyEmployees) {
    for (const employee of affectedEmployees) {
      await smsService.send({
        to: employee.phone,
        message: `Schedule Update: Your shifts have been changed. Reason: ${reason}. Check app for updated schedule.`
      });
      
      await emailService.send({
        to: employee.email,
        subject: 'Schedule Changed',
        body: `Your schedule has been updated.
        
Old shifts:
${employee.oldShifts.map(s => `${s.date} ${s.time} at ${s.site}`).join('\n')}

New shifts:
${employee.newShifts.map(s => `${s.date} ${s.time} at ${s.site}`).join('\n')}

Reason: ${reason}`
      });
    }
  }
  
  // 7. Trinity logs rollback
  await trinity.log({
    action: 'schedule.rollback',
    scheduleId,
    reason,
    affectedEmployees: affectedEmployees.length,
    performedBy: userId
  });
  
  return {
    success: true,
    affectedEmployees: affectedEmployees.length,
    restoredScheduleId: newSchedule.id
  };
}
