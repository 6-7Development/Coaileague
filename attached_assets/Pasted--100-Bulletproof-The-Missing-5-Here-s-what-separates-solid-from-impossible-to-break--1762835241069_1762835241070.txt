## **1. Race-Condition-Proof Autosave**

Your current `useCheckpoint` saves on every `setData`, but **doesn't debounce or handle server conflicts**. Add:

```ts
// ui/hooks/useCheckpoint.ts (upgraded)
import { useEffect, useRef, useState, useCallback } from "react";
import { debounce } from "lodash"; // or custom debounce

export function useCheckpoint(key: string, initial: any = {}, serverSync = false) {
  const [data, setData] = useState<any>(() => {
    try {
      const s = sessionStorage.getItem(key);
      return s ? JSON.parse(s) : initial;
    } catch {
      return initial;
    }
  });

  const dirty = useRef(false);
  const saveVersion = useRef(0); // prevent stale writes

  // Client-side save (immediate)
  useEffect(() => {
    sessionStorage.setItem(key, JSON.stringify(data));
    dirty.current = true;
  }, [key, data]);

  // Server-side save (debounced, versioned)
  const syncToServer = useCallback(
    debounce(async (payload: any, version: number) => {
      if (version !== saveVersion.current) return; // stale
      try {
        await fetch("/api/drafts", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ key, payload }),
        });
      } catch (err) {
        console.error("Draft sync failed:", err);
      }
    }, 1000),
    [key]
  );

  useEffect(() => {
    if (serverSync) {
      saveVersion.current++;
      syncToServer(data, saveVersion.current);
    }
  }, [data, serverSync, syncToServer]);

  // Block navigation if dirty
  useEffect(() => {
    const onBefore = (e: BeforeUnloadEvent) => {
      if (dirty.current) {
        e.preventDefault();
        e.returnValue = "";
      }
    };
    window.addEventListener("beforeunload", onBefore);
    return () => window.removeEventListener("beforeunload", onBefore);
  }, []);

  return {
    data,
    setData,
    clear: () => {
      sessionStorage.removeItem(key);
      dirty.current = false;
      if (serverSync) fetch(`/api/drafts?key=${key}`, { method: "DELETE" });
    },
  };
}
```

**Why this matters**: Without versioning, a fast user can trigger **overlapping saves** that overwrite each other. The `saveVersion.current` check guarantees only the latest write lands.

## **2. Offline-First with Service Worker**

Right now, if the network drops mid-save, **data disappears**. Add a service worker to queue failed requests:

```ts
// public/sw.js
const CACHE_NAME = "v1";
const OFFLINE_QUEUE = "offline-queue";

self.addEventListener("fetch", (event) => {
  if (event.request.method === "POST" && event.request.url.includes("/api/drafts")) {
    event.respondWith(
      fetch(event.request.clone()).catch(async () => {
        // Network failed—queue the request
        const cache = await caches.open(OFFLINE_QUEUE);
        await cache.put(event.request.url, new Response(await event.request.text()));
        return new Response(JSON.stringify({ queued: true }), { status: 202 });
      })
    );
  }
});

// Replay queue when online
self.addEventListener("sync", (event) => {
  if (event.tag === "replay-drafts") {
    event.waitUntil(replayQueue());
  }
});

async function replayQueue() {
  const cache = await caches.open(OFFLINE_QUEUE);
  const requests = await cache.keys();
  for (const req of requests) {
    const body = await (await cache.match(req)).text();
    await fetch(req.url, { method: "POST", body, headers: { "Content-Type": "application/json" } });
    await cache.delete(req);
  }
}
```

Register it:

```tsx
// index.tsx or _app.tsx
if ("serviceWorker" in navigator) {
  navigator.serviceWorker.register("/sw.js");
}
```

## **3. Server-Side Conflict Resolution**

If two users edit the same draft simultaneously, **last-write-wins loses data**. Add optimistic locking:

```sql
-- Add version column to drafts table
ALTER TABLE drafts ADD COLUMN version INT DEFAULT 1;
```

```ts
// utils/serverDrafts.ts (with conflict detection)
export async function saveDraft(
  orgId: string,
  userId: string,
  scope: string,
  key: string,
  payload: any,
  expectedVersion?: number
) {
  const result = await pool.query(
    `INSERT INTO drafts(org_id, user_id, scope, key, payload, version, updated_at) 
     VALUES($1, $2, $3, $4, $5, 1, now())
     ON CONFLICT(org_id, user_id, scope, key) 
     DO UPDATE SET 
       payload = CASE 
         WHEN drafts.version = $6 OR $6 IS NULL THEN $5 
         ELSE drafts.payload 
       END,
       version = drafts.version + 1,
       updated_at = now()
     RETURNING version, (version = $6 + 1 OR $6 IS NULL) as saved`,
    [orgId, userId, scope, key, JSON.stringify(payload), expectedVersion ?? null]
  );

  if (!result.rows[0]?.saved) {
    throw new Error("CONFLICT: Draft was modified by another session");
  }

  return result.rows[0].version;
}
```

**Client handles conflict**:

```tsx
try {
  const newVersion = await saveDraft(orgId, userId, scope, key, data, currentVersion);
  setCurrentVersion(newVersion);
} catch (err) {
  if (err.message.includes("CONFLICT")) {
    // Show merge UI or force-reload
    alert("Someone else edited this. Reloading...");
    location.reload();
  }
}
```

## **5. 404 Logging with Deduplication**

Your middleware logs **every 404**, but bots/scanners will flood `web_404_events`. Add:

```ts
// middleware/catch-all-404.ts (deduplicated)
const recentPaths = new Map<string, number>(); // path -> lastSeen timestamp

export function catchAll404() {
  return async (req: any, res: any, next: any) => {
    const now = Date.now();
    const key = `${req.method}:${req.path}`;
    
    // Only log if not seen in last 5 minutes
    if (!recentPaths.has(key) || now - recentPaths.get(key)! > 300000) {
      recentPaths.set(key, now);
      try {
        await pool.query(
          `INSERT INTO web_404_events(path, method, user_id, org_id, referer, ua) 
           VALUES($1,$2,$3,$4,$5,$6)`,
          [req.path, req.method, req.user?.id || null, req.user?.org_id || null, 
           req.get("referer") || null, req.get("user-agent") || null]
        );
      } catch {}
    }
    
    res.status(404).redirect("/safe-fallback?from=404");
  };
}
```

## **6. Deep-Link Restore (Mobile/PWA)**

If a user gets a link like `/dashboard/project/123/edit` while offline, they hit a 404. **Cache the intent**:

```tsx
// App.tsx or router setup
useEffect(() => {
  const intended = sessionStorage.getItem("intended_url");
  if (intended && navigator.onLine) {
    sessionStorage.removeItem("intended_url");
    location.assign(intended);
  }
}, []);

// In 404 handler
if (!navigator.onLine) {
  sessionStorage.setItem("intended_url", req.path);
}
```

// routes/health.ts
import { Router } from "express";

const r = Router();

r.get("/health", async (req, res) => {
