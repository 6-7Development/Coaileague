## Implementation Tasks

### 1. Add Backup AI Providers (Keep Gemini Primary)

```javascript
// Add this wrapper around all Gemini API calls

async function callAIWithFallback(prompt, context) {
  const providers = [
    { name: 'Gemini', call: () => callGemini(prompt, context) },
    { name: 'Claude', call: () => callClaude(prompt, context) },
    { name: 'GPT-4', call: () => callGPT4(prompt, context) }
  ];

  for (const provider of providers) {
    try {
      console.log(`Attempting ${provider.name}...`);
      const response = await provider.call();
      
      // Log which provider succeeded
      await logToAudit({
        event: 'ai_call_success',
        provider: provider.name,
        timestamp: new Date()
      });
      
      return response;
      
    } catch (error) {
      console.error(`${provider.name} failed:`, error);
      
      await logToAudit({
        event: 'ai_call_failure',
        provider: provider.name,
        error: error.message,
        timestamp: new Date()
      });
      
      // Try next provider
      continue;
    }
  }
  
  // All providers failed - enter degraded mode
  throw new Error('All AI providers unavailable');
}
```

**What to do:**
- Keep all existing Gemini integration exactly as-is
- Replace direct Gemini calls with `callAIWithFallback()`
- Add API credentials for Claude and GPT-4 as backups
- Gemini stays primary - backups only activate on failure

### 2. Enhanced State Persistence

**Expand the Audit Ledger to capture workflow state:**

```javascript
// When Trinity starts any task
async function saveWorkflowCheckpoint(workflowData) {
  await auditLedger.save({
    type: 'workflow_checkpoint',
    workflow_id: workflowData.id,
    current_step: workflowData.step,
    context: workflowData.context,
    pending_actions: workflowData.pendingActions,
    timestamp: new Date(),
    status: 'in_progress'
  });
}

// On system restart, check for incomplete work
async function recoverIncompleteWorkflows() {
  const incomplete = await auditLedger.find({
    type: 'workflow_checkpoint',
    status: 'in_progress'
  });
  
  for (const workflow of incomplete) {
    console.log(`Resuming workflow: ${workflow.workflow_id}`);
    await trinity.resume(workflow);
  }
}
```

**What to do:**
- Before each major Trinity action, save current state
- On app startup, run `recoverIncompleteWorkflows()`
- Mark workflows as 'completed' when finished
- Trinity can now pick up exactly where she left off

### 3. Critical Function Fallbacks

**For payroll, scheduling, and time tracking - add simple rule-based backups:**

```javascript
// Example: Payroll processing with fallback
async function processPayroll(employees) {
  try {
    // Try AI-enhanced processing (Gemini primary)
    return await trinity.optimizePayroll(employees);
    
  } catch (aiError) {
    console.warn('AI unavailable, using rule-based payroll');
    
    // Simple rule-based calculation
    return employees.map(emp => ({
      employee_id: emp.id,
      hours: emp.hoursWorked,
      rate: emp.hourlyRate,
      gross_pay: emp.hoursWorked * emp.hourlyRate,
      deductions: calculateStandardDeductions(emp),
      net_pay: calculateNetPay(emp)
    }));
  }
}

// Example: Simple scheduling fallback
async function createSchedule(team, requirements) {
  try {
    return await trinity.optimizeSchedule(team, requirements);
    
  } catch (aiError) {
    console.warn('AI unavailable, using template schedule');
    return generateTemplateSchedule(team, requirements);
  }
}
```

### 4. User Notifications

**Let users know when running in backup mode:**

```javascript
// Add to UI state management
const systemStatus = {
  aiProvider: 'Gemini', // or 'Claude' or 'GPT-4'
  mode: 'normal', // or 'degraded'
  lastCheck: new Date()
};

// Show banner when in backup mode
function StatusBanner() {
  if (systemStatus.mode === 'degraded') {
    return (
      <div className="bg-yellow-100 border-yellow-400 p-4">
        ⚠️ Running in backup mode - some AI features limited
      </div>
    );
  }
  
  if (systemStatus.aiProvider !== 'Gemini') {
    return (
      <div className="bg-blue-100 border-blue-400 p-4">
        ℹ️ Using backup AI provider: {systemStatus.aiProvider}
      </div>
    );
  }
  
  return null;
}
```
