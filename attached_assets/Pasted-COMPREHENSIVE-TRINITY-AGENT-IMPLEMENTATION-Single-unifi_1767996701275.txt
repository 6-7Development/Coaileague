2. CREATE: /lib/trinity/streaming/websocket-service.ts
   Purpose: Real-time streaming to UI
   Code structure:
   
   import { WebSocket } from 'ws';
   
   class WebSocketService {
     private connections = new Map<string, WebSocket>();
     
     connect(conversationId: string, ws: WebSocket) {
       this.connections.set(conversationId, ws);
     }
     
     async emit(conversationId: string, event: string, data: any) {
       const ws = this.connections.get(conversationId);
       if (ws && ws.readyState === WebSocket.OPEN) {
         ws.send(JSON.stringify({ event, data, timestamp: Date.now() }));
       }
     }
     
     disconnect(conversationId: string) {
       this.connections.delete(conversationId);
     }
   }
   
   export const websocketService = new WebSocketService();

    import { useEffect, useState } from 'react';
    
    export const useWebSocket = (eventType: string, callback?: (data: any) => void) => {
      const [data, setData] = useState(null);
      const [ws, setWs] = useState<WebSocket | null>(null);
      
      useEffect(() => {
        const socket = new WebSocket(`/api/trinity/ws/${conversationId}`);
        
        socket.onmessage = (event) => {
          const parsed = JSON.parse(event.data);
          if (parsed.event === eventType) {
            setData(parsed.data);
            callback?.(parsed.data);
          }
        };
        
        setWs(socket);
        return () => socket.close();
      }, [eventType]);
      
      return data;
    };
