typescript// In server.ts, update callGeminiAPI function:
import { generateContent } from '../finops/middleware';

async function callGeminiAPI(message, context) {
  const result = await generateContent(workspaceId, message, {
    trackValue: true,
    valueEventType: 'INSIGHT_GENERATED',
  });
  
  return {
    response: result.text,
    confidence: 0.85,
  };
}
