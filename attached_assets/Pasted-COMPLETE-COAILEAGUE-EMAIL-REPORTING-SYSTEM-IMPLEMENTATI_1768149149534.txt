javascript// LOCATION: /app/api/reports/client-delivery.js

const ClientDelivery = {
  async sendToClient(report, client) {
    // Format report as PDF
    const pdf = await generatePDF(report);
    
    // Prepare email
    const email = {
      to: client.contacts,
      subject: `${report.type} Report - ${report.location} - ${report.date}`,
      body: generateEmailBody(report),
      attachments: [pdf],
      
      // For urgent/critical
      priority: report.priority === 'critical' ? 'high' : 'normal',
      sms_alert: report.priority === 'critical'
    };
    
    // Send via Resend
    await sendViaResend(email);
    
    // Log delivery
    await logDelivery(report, client);
    
    // Update client portal
    await updateClientPortal(report);
  },
  
  clientPortalView: {
    // Reports dashboard for clients
    - Filter by date range
    - Search by incident type
    - Download reports
    - Analytics dashboard
    - Trend analysis
  }
};
