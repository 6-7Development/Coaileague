# CoAIleague Invoice Data Architecture & Automation Spec

## Overview

CoAIleague must support complex billing scenarios where:
- A **Primary Client** (e.g., "Off Duty Officers") has multiple **Sub-Clients** (e.g., "Dura Line LLC")
- Each Sub-Client has their own job sites, bill rates, and job IDs
- Invoices must include ALL relevant data for proper billing
- Time tracking must flow correctly to payroll AND invoicing
- QuickBooks sync must maintain full parity with all nested data

**CRITICAL:** No money can be lost due to missing data, schema errors, or incomplete automation.

---

## Data Model

### 1. Clients Table (Primary Clients)

```sql

  workspace_id UUID REFERENCES workspaces(id),
  
  -- Basic Info

  billing_state VARCHAR(50),                     -- "CA"

  shipping_city VARCHAR(100),

  shipping_zip VARCHAR(20),
```

## Rate Lookup Logic

When creating a shift or time entry, rates are looked up in this order:

### Bill Rate (What to charge client)

```javascript
async function getBillRate(employeeId, clientId, subClientId, siteId) {
  // 1. Check site-specific rate
  if (siteId) {
    const site = await db.sites.findById(siteId);
    if (site.bill_rate) return site.bill_rate;
  }
  
  // 2. Check sub-client rate
  if (subClientId) {
    const subClient = await db.subClients.findById(subClientId);
    if (subClient.bill_rate) return subClient.bill_rate;
  }
  
  // 3. Check client rate
  if (clientId) {
    const client = await db.clients.findById(clientId);
    if (client.default_bill_rate) return client.default_bill_rate;
  }
  
  // 4. Workspace default
  const workspace = await db.workspaces.findById(workspaceId);
  return workspace.default_bill_rate;
}
```

### Pay Rate (What to pay employee)

```javascript
async function getPayRate(employeeId, clientId, subClientId, siteId) {
  // 1. Check employee-specific rate for this site
  const siteRate = await db.employeeClientRates.findOne({
    employee_id: employeeId,
    site_id: siteId
  });
  if (siteRate) return siteRate.pay_rate;
  
  // 2. Check employee-specific rate for this sub-client
  const subClientRate = await db.employeeClientRates.findOne({
    employee_id: employeeId,
    sub_client_id: subClientId
  });
  if (subClientRate) return subClientRate.pay_rate;
  
  // 3. Check employee-specific rate for this client
  const clientRate = await db.employeeClientRates.findOne({
    employee_id: employeeId,
    client_id: clientId
  });
  if (clientRate) return clientRate.pay_rate;
  
  // 4. Employee default rate
  const employee = await db.employees.findById(employeeId);
  return employee.default_pay_rate;
}
```

## Invoice Generation Automation

### Trinity Auto-Invoice Generation

```javascript
async function generateInvoiceForClient(clientId, periodStart, periodEnd) {
  // 1. Get all approved time entries for this client in the period
  const timeEntries = await db.timeEntries.find({
    client_id: clientId,
    status: 'approved',
    clock_in_at: { $gte: periodStart, $lte: periodEnd },
    invoice_id: null  // Not yet invoiced
  });
  
  if (timeEntries.length === 0) {
    return { success: false, message: 'No approved time entries to invoice' };
  }
  
  // 2. Group time entries by sub-client
  const groupedBySubClient = groupBy(timeEntries, 'sub_client_id');
  
  // 3. Create invoice
  const invoice = await db.invoices.create({
    workspace_id: workspaceId,
    client_id: clientId,
    invoice_number: generateInvoiceNumber(),
    invoice_date: new Date(),
    due_date: calculateDueDate(client.default_payment_terms),
    service_period_start: periodStart,
    service_period_end: periodEnd,
    // ... populate billing addresses from client
  });
  
  // 4. Create line items for each sub-client
  let lineNumber = 1;
  let subtotal = 0;
  
  for (const [subClientId, entries] of Object.entries(groupedBySubClient)) {
    const subClient = await db.subClients.findById(subClientId);
    
    // Calculate totals for this sub-client
    const totalHours = entries.reduce((sum, e) => sum + e.regular_hours + e.overtime_hours, 0);
    const totalAmount = entries.reduce((sum, e) => sum + e.billable_amount, 0);
    
    // Build description with all dates
    const serviceDates = entries.map(e => ({
      date: formatDate(e.clock_in_at),
      time: `${formatTime(e.clock_in_at)}-${formatTime(e.clock_out_at)}`
    }));
    
    // Get unique employees
    const employeeIds = [...new Set(entries.map(e => e.employee_id))];
    const employees = await db.employees.findByIds(employeeIds);
    const officerNames = employees.map(e => `${e.first_name} ${e.last_name}`);
    
    // Build structured description
    const descriptionData = {
      job_id: subClient.client_job_id,
      sub_client_name: subClient.name,
      location: formatAddress(subClient),
      schedule_description: subClient.default_schedule_description,
      service_dates: serviceDates,
      officers: officerNames
    };
    
    // Build text description (for display)
    const description = buildInvoiceDescription(descriptionData);
    
    // Create line item
    await db.invoiceLineItems.create({
      invoice_id: invoice.id,
      line_number: lineNumber++,
      service_date: periodStart,
      product_service_name: `Armed:${subClient.name} ODO Armad`,
      sub_client_id: subClientId,
      description: description,
      description_data: descriptionData,
      quantity: totalHours,
      rate: subClient.bill_rate,
      amount: totalAmount,
      time_entry_ids: entries.map(e => e.id)
    });
    
    subtotal += totalAmount;
    
    // Mark time entries as invoiced
    await db.timeEntries.updateMany(
      { id: { $in: entries.map(e => e.id) } },
      { invoice_id: invoice.id, status: 'invoiced' }
    );
  }
  
  // 5. Calculate totals
  const taxAmount = client.tax_exempt ? 0 : subtotal * (client.tax_rate / 100);
  const totalAmount = subtotal + taxAmount;
  
  await db.invoices.update(invoice.id, {
    subtotal,
    tax_amount: taxAmount,
    total_amount: totalAmount,
    balance_due: totalAmount
  });
  
  return { success: true, invoice };
}

function buildInvoiceDescription(data) {
  let desc = '';
  
  desc += `Job ID: ${data.job_id}\n\n`;
  desc += `Client: ${data.sub_client_name}\n`;
  desc += `Location: ${data.location}\n\n`;
  desc += `Hours: ${data.schedule_description}\n`;
  
  // Add date list
  for (const sd of data.service_dates) {
    desc += `${sd.date} ${sd.time}\n`;
  }
  
  // Add officers if required
  if (data.officers && data.officers.length > 0) {
    desc += `\nOfficers: ${data.officers.join(', ')}`;
  }
  
  return desc;
}
```

## QuickBooks Sync

### Invoice Sync to QuickBooks

```javascript
async function syncInvoiceToQuickBooks(invoiceId) {
  const invoice = await db.invoices.findById(invoiceId);
  const client = await db.clients.findById(invoice.client_id);
  const lineItems = await db.invoiceLineItems.find({ invoice_id: invoiceId });
  
  // Ensure client exists in QB
  if (!client.qb_customer_id) {
    await syncClientToQuickBooks(client.id);
  }
  
  // Build QB invoice object
  const qbInvoice = {
    CustomerRef: { value: client.qb_customer_id },
    DocNumber: invoice.invoice_number,
    TxnDate: formatDateForQB(invoice.invoice_date),
    DueDate: formatDateForQB(invoice.due_date),
    BillEmail: { Address: invoice.bill_to_email },
    Line: []
  };
  
  // Add line items
  for (const item of lineItems) {
    // Get or create QB product/service item
    const qbItemId = await getOrCreateQBServiceItem(item.product_service_name);
    
    qbInvoice.Line.push({
      DetailType: 'SalesItemLineDetail',
      Amount: item.amount,
      Description: item.description,
      SalesItemLineDetail: {
        ItemRef: { value: qbItemId },
        Qty: item.quantity,
        UnitPrice: item.rate
      }
    });
  }
  
  // Create or update in QB
  let qbResponse;
  if (invoice.qb_invoice_id) {
    // Update existing
    qbResponse = await qb.updateInvoice(invoice.qb_invoice_id, qbInvoice);
  } else {
    // Create new
    qbResponse = await qb.createInvoice(qbInvoice);
  }
  
  // Store QB reference
  await db.invoices.update(invoiceId, {
    qb_invoice_id: qbResponse.Id,
    qb_synced_at: new Date()
  });
  
  return { success: true, qbInvoiceId: qbResponse.Id };
}
```

```javascript
async function syncSubClientToQuickBooks(subClientId) {

    Job: true,  // Mark as sub-customer/job
    BillAddr: {
      Line1: subClient.location_address_line1,
      City: subClient.location_city,
      CountrySubDivisionCode: subClient.location_state,
      PostalCode: subClient.location_zip
    }
  };
  
  let qbResponse;
  if (subClient.qb_sub_customer_id) {
    qbResponse = await qb.updateCustomer(subClient.qb_sub_customer_id, qbSubCustomer);
  } else {
    qbResponse = await qb.createCustomer(qbSubCustomer);
  }
  
  await db.subClients.update(subClientId, {
    qb_sub_customer_id: qbResponse.Id,
    qb_synced_at: new Date()
  });
  
  return { success: true, qbSubCustomerId: qbResponse.Id };
}
```

### Before Invoice Generation, Verify:

```javascript
async function validateInvoiceData(clientId, periodStart, periodEnd) {
  const errors = [];
  const warnings = [];
  
  // 1. Client has billing info
  const client = await db.clients.findById(clientId);
  if (!client.billing_address_line1) errors.push('Client missing billing address');
  if (!client.email) warnings.push('Client missing email');
  
  // 2. All time entries have required data
  const timeEntries = await db.timeEntries.find({
    client_id: clientId,
    status: 'approved',
    clock_in_at: { $gte: periodStart, $lte: periodEnd }
  });
  
  for (const entry of timeEntries) {
    if (!entry.sub_client_id) {
      errors.push(`Time entry ${entry.id} missing sub_client_id`);
    }
    if (!entry.bill_rate) {
      errors.push(`Time entry ${entry.id} missing bill_rate`);
    }
    if (!entry.billable_amount) {
      errors.push(`Time entry ${entry.id} missing billable_amount`);
    }
  }
  
  // 3. All sub-clients have required data
  const subClientIds = [...new Set(timeEntries.map(e => e.sub_client_id))];
  for (const subClientId of subClientIds) {
    const subClient = await db.subClients.findById(subClientId);
    if (!subClient.client_job_id) {
      warnings.push(`Sub-client ${subClient.name} missing job ID`);
    }
    if (!subClient.location_address_line1) {
      warnings.push(`Sub-client ${subClient.name} missing location address`);
    }
    if (!subClient.bill_rate) {
      errors.push(`Sub-client ${subClient.name} missing bill rate`);
    }
  }
  
  return { valid: errors.length === 0, errors, warnings };
}
```
